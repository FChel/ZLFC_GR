method assets_create_entity.
*-----------------------------------------------------------------------------------------
* DESCRIPTION:  Assets Create Entity - saves update to asset data
*
* AUTHOR:       Fred Chelashaw
* CREATE DATE:  30.11.2020
*
* ACTIVITY:     AD-11-32-10 - MyFi Purchase Receipting
* DEV SCRIPT:   DS-11-32-10-01 - MyFi Purchase Receipting App
*-----------------------------------------------------------------------------------------
* VERSION CONTROL:
*
* Date       Author    CTS Req    SCR        Description
* ---------- --------- ---------- ---------- ---------------------------------------------
* 30.11.2020 220977FKC RD2K920449 6000002172 MyFi Purchasing (Release 2.3)
*-----------------------------------------------------------------------------------------

  data:
    lo_message_container  type ref to /iwbep/if_message_container,
    ls_generaldata        type bapi1022_feglg001,
    ls_generaldatax       type bapi1022_feglg001x,
    ls_inventory          type bapi1022_feglg011,
    ls_inventoryx	        type bapi1022_feglg011x,
    ls_timedependentdata  type bapi1022_feglg003,
    ls_timedependentdatax type bapi1022_feglg003x,
    l_msgv                type symsgv,
    ls_return             type bapiret2.

*===> read the supplied data
  io_data_provider->read_entry_data(
    importing
      es_data = er_entity ).

*--- message container setup
  lo_message_container = mo_context->get_message_container( ).

*--- convert to BAPI format
  if er_entity-serial_no_off ne abap_true.
    ls_generaldata-serial_no = er_entity-serial_no.
    ls_generaldatax-serial_no = abap_true.
  endif.

  if er_entity-invent_no_off ne abap_true.
    ls_generaldata-invent_no = er_entity-invent_no.
    ls_generaldatax-invent_no = abap_true.
  endif.

  if er_entity-main_descript_off ne abap_true.
    ls_generaldata-main_descript = er_entity-main_descript.
    ls_generaldatax-main_descript = abap_true.
    ls_generaldata-descript = er_entity-main_descript.
    ls_generaldatax-descript = abap_true.
  endif.

  if er_entity-note_off ne abap_true.
    ls_inventory-note = er_entity-note.
    ls_inventoryx-note = abap_true.
  endif.

  ls_timedependentdata-from_date = sy-datum.
  ls_timedependentdata-to_date = '99991231'.

  if er_entity-plate_no_off ne abap_true.
    ls_timedependentdata-plate_no = er_entity-plate_no.
    ls_timedependentdatax-license_plate_no = abap_true.
  endif.

  if er_entity-room_off ne abap_true
  and er_entity-room_locked ne abap_true.
    ls_timedependentdata-room = er_entity-room.
    ls_timedependentdatax-room = abap_true.
  endif.

*--- update asset using BAPI
  call function 'BAPI_FIXEDASSET_CHANGE'
    exporting
      companycode        = er_entity-comp_code
      asset              = er_entity-asset_no
      subnumber          = er_entity-sub_number
      generaldata        = ls_generaldata
      generaldatax       = ls_generaldatax
      inventory          = ls_inventory
      inventoryx         = ls_inventoryx
      timedependentdata  = ls_timedependentdata
      timedependentdatax = ls_timedependentdatax
    importing
      return             = ls_return.

  if sy-subrc = 0 and ls_return-type = rs_c_success.
    call function 'BAPI_TRANSACTION_COMMIT'
      exporting
        wait   = abap_true
      importing
        return = ls_return.
    if sy-subrc = 0.
      call function 'CONVERSION_EXIT_ALPHA_OUTPUT'
        exporting
          input  = er_entity-asset_no
        importing
          output = er_entity-asset_no.
      call function 'CONVERSION_EXIT_ALPHA_OUTPUT'
        exporting
          input  = er_entity-sub_number
        importing
          output = er_entity-sub_number.
      concatenate er_entity-asset_no '-' er_entity-sub_number into l_msgv
        separated by space.
      "l_msgv = er_entity-asset_no && ' - ' && er_entity-sub_number.
      message s012 into ls_return-message.
      lo_message_container->add_message(
        iv_msg_type = sy-msgty
        iv_msg_id = sy-msgid
        iv_msg_number = sy-msgno
        iv_msg_v1 = l_msgv
        iv_add_to_response_header = abap_true ).
    else.
      lo_message_container->add_message(
        iv_msg_type = sy-msgty
        iv_msg_id = sy-msgid
        iv_msg_number = sy-msgno
        iv_msg_v1 = sy-msgv1
        iv_msg_v2 = sy-msgv2
        iv_msg_v3 = sy-msgv3
        iv_msg_v4 = sy-msgv4
        iv_add_to_response_header = abap_true ).
    endif.
  else.
    if ls_return is not initial.
      lo_message_container->add_message_from_bapi(
        is_bapi_message = ls_return
        iv_add_to_response_header = abap_true
        iv_message_target = '' ).
    else.
      lo_message_container->add_message(
        iv_msg_type = sy-msgty
        iv_msg_id = sy-msgid
        iv_msg_number = sy-msgno
        iv_msg_v1 = sy-msgv1
        iv_msg_v2 = sy-msgv2
        iv_msg_v3 = sy-msgv3
        iv_msg_v4 = sy-msgv4
        iv_add_to_response_header = abap_true ).
    endif.
  endif.

endmethod.

method assets_get_entity.
*-----------------------------------------------------------------------------------------
* DESCRIPTION:  Assets Get Entity - returns asset details
*
* AUTHOR:       Fred Chelashaw
* CREATE DATE:  30.11.2020
*
* ACTIVITY:     AD-11-32-10 - MyFi Purchase Receipting
* DEV SCRIPT:   DS-11-32-10-01 - MyFi Purchase Receipting App
*-----------------------------------------------------------------------------------------
* VERSION CONTROL:
*
* Date       Author    CTS Req    SCR        Description
* ---------- --------- ---------- ---------- ---------------------------------------------
* 30.11.2020 220977FKC RD2K920449 6000002172 MyFi Purchasing (Release 2.3)
*-----------------------------------------------------------------------------------------

  data:
    l_felei type felei.

*--- get converted key
  io_tech_request_context->get_converted_keys(
    importing
      es_key_values = er_entity ).

*--- get asset detail for selected key
  select single anla~anlkl as assetclass
      anla~invzu as note
      anla~invnr as invent_no
      anla~sernr as serial_no
      anlh~anlhtxt as main_descript

method _cancel_gr.
*-----------------------------------------------------------------------------------------
* DESCRIPTION:  Helper Method to cancel a Goods Receipt
*
* AUTHOR:       Fred Chelashaw
* CREATE DATE:  13.04.2020
*
* ACTIVITY:     AD-11-32-10 - MyFi Purchase Receipting
* DEV SCRIPT:   DS-11-32-10-01 - MyFi Purchase Receipting App
*-----------------------------------------------------------------------------------------
* VERSION CONTROL:
*
* Date       Author    CTS Req    SCR        Description
* ---------- --------- ---------- ---------- ---------------------------------------------
* 13.04.2020 220977FKC RD2K919103 6000001870 DFG BAR 2020-02 MyFi Purchase Receipting
*-----------------------------------------------------------------------------------------

  data:
    lo_message_container type ref to /iwbep/if_message_container,
    lx_root              type ref to cx_root,
    lt_return            type bapiret2_t,
    ls_return            like line of lt_return,
    lt_gr_item           type standard table of bapi2017_gm_item_04,
    ls_gr_item           like line of lt_gr_item,
    ls_result            type bapi2017_gm_head_ret,
    l_message            type bapi_msg,
    ls_goodsreceiptitem  type zcl_zlfc_gr_mpc=>ts_goodsreceiptitem,
    lt_goodsreceiptitem  type zcl_zlfc_gr_mpc=>tt_goodsreceiptitem,
    ls_materialdocument  type zcl_zlfc_gr_mpc=>ts_materialdocument.

  field-symbols:
    <ls_goodsreceiptitem>  type zcl_zlfc_gr_mpc=>ts_goodsreceiptitem.

*--- message container setup
  lo_message_container = mo_context->get_message_container( ).

*--- get gr items
  loop at cs_goodsreceipt_deep-items into ls_goodsreceiptitem
  where selected = abap_true.
    clear ls_gr_item.
    ls_gr_item-matdoc_item = ls_goodsreceiptitem-matdoc_itm.
    append ls_gr_item to lt_gr_item.
    append ls_goodsreceiptitem to lt_goodsreceiptitem.
  endloop.

*--- cancel the gr
  call function 'BAPI_GOODSMVT_CANCEL'
    exporting
      materialdocument    = cs_goodsreceipt_deep-mat_doc
      matdocumentyear     = cs_goodsreceipt_deep-doc_year
    importing
      goodsmvt_headret    = ls_result
    tables
      return              = lt_return
      goodsmvt_matdocitem = lt_gr_item.

  if lt_return is not initial.

*--- replace WBS/IO system message with custom
    loop at lt_return into ls_return where type = 'E'.

*--- read PO item number & remove leading zeros (for custom error messages)
      clear ls_goodsreceiptitem.
      read table lt_goodsreceiptitem into ls_goodsreceiptitem index ls_return-row.
      call function 'CONVERSION_EXIT_ALPHA_OUTPUT'
        exporting
          input  = ls_goodsreceiptitem-po_item
        importing
          output = ls_goodsreceiptitem-po_item.
      concatenate text-pp1 ls_goodsreceiptitem-po_item into l_message separated by space.

*--- replace wbs/io error
      if ls_return-id = 'BS' and ls_return-number = 013.
        delete lt_return index sy-tabix.

        if ls_return-message_v2(3) = 'ORD'.
          replace first occurrence of 'ORD' in ls_return-message_v2 with text-aa4.
        endif.

        message e009 with ls_return-message_v2 l_message ac_ap_name ac_ap_email into l_message.
        lo_message_container->add_message(
          iv_msg_type = sy-msgty
         iv_msg_id = sy-msgid
          iv_msg_number = sy-msgno
          iv_msg_v1 = sy-msgv1
          iv_msg_v2 = sy-msgv2
          iv_msg_v3 = sy-msgv3
          iv_msg_v4 = sy-msgv4
          iv_add_to_response_header = abap_true ).
      endif.

*--- replace gl error
      if ls_return-id = 'M7' and ( ls_return-number = 161 or ls_return-number = 160 ).
        delete lt_return index sy-tabix.

        concatenate text-aa1 ls_return-message_v1 into ls_return-message_v1 separated by space.
        message e009 with ls_return-message_v1 l_message ac_ap_name ac_ap_email into l_message.
        lo_message_container->add_message(
          iv_msg_type = sy-msgty
          iv_msg_id = sy-msgid
          iv_msg_number = sy-msgno
          iv_msg_v1 = sy-msgv1
          iv_msg_v2 = sy-msgv2
          iv_msg_v3 = sy-msgv3
          iv_msg_v4 = sy-msgv4
          iv_add_to_response_header = abap_true ).
      endif.

*--- replace cc error
      if ls_return-id = 'KI' and ls_return-number = 260.
        delete lt_return index sy-tabix.
        concatenate text-aa2 ls_return-message_v2 into ls_return-message_v1 separated by space.
        message e009 with ls_return-message_v1 l_message ac_ap_name ac_ap_email into l_message.
        lo_message_container->add_message(
          iv_msg_type = sy-msgty
          iv_msg_id = sy-msgid
          iv_msg_number = sy-msgno
          iv_msg_v1 = sy-msgv1
          iv_msg_v2 = sy-msgv2
          iv_msg_v3 = sy-msgv3
          iv_msg_v4 = sy-msgv4
          iv_add_to_response_header = abap_true ).
      endif.

*--- highlight line
      read table cs_goodsreceipt_deep-items assigning <ls_goodsreceiptitem>
        with key matdoc_itm = ls_goodsreceiptitem-matdoc_itm.
      if sy-subrc = 0.
        <ls_goodsreceiptitem>-line_status = 'Error'.
      endif.

    endloop.

    lo_message_container->add_messages_from_bapi(
      it_bapi_messages = lt_return
      iv_add_to_response_header = abap_true ).

  elseif ls_result-mat_doc is not initial.

    call function 'BAPI_TRANSACTION_COMMIT'
      exporting
        wait   = abap_true
      importing
        return = ls_return.

    if ls_return is not initial.
      lo_message_container->add_message_from_bapi(
        is_bapi_message = ls_return
        iv_add_to_response_header = abap_true
        iv_message_target = '' ).
    else.
      ls_materialdocument-mat_doc = ls_result-mat_doc.
      ls_materialdocument-doc_year = ls_result-doc_year.
      append ls_materialdocument to cs_goodsreceipt_deep-materialdocuments.
    endif.

  endif.

endmethod.

      matdocumentyear  = i_doc_year
    importing
      goodsmvt_header  = ls_header
    tables
      goodsmvt_items   = lt_item_show
      return           = lt_return.
  if lt_return is not initial.
    lo_message_container->add_messages_from_bapi(
      it_bapi_messages = lt_return
      iv_add_to_response_header = abap_true ).

    rs_goodsreceipt_deep-doc_year = '0001'.
  else.
*--- header details
    rs_goodsreceipt_deep-mat_doc = i_mat_doc.
    rs_goodsreceipt_deep-doc_year = i_doc_year.
    rs_goodsreceipt_deep-username = ls_header-username.

*--- lookup full name
  select single name_text
    from v_usr_name
    into @data(l_name_text)
    where bname = @ls_header-username.
  if l_name_text is not initial.
    rs_goodsreceipt_deep-username_full = l_name_text.
  else.
    rs_goodsreceipt_deep-username_full = ls_header-username.
  endif.

*--- read mseg for fields not returned by bapi
    select zeile shkzg
      from mseg
      into corresponding fields of table lt_mseg
      where mblnr = i_mat_doc
        and mjahr = i_doc_year.

*--- prepare items ... for gr lines that relate to the same po line item these
*    get combined into one so that the history checking logic below works correctly
*    otherwise the balance check is per split and is not always correct
    sort lt_item_show by mat_doc doc_year po_number po_item matdoc_itm.

    loop at lt_item_show into ls_item.
      read table lt_item assigning <ls_item> with table key
          mat_doc = ls_item-mat_doc
          doc_year = ls_item-doc_year
          po_number = ls_item-po_number
          po_item = ls_item-po_item.
      if sy-subrc = 0.
        add ls_item-entry_qnt to <ls_item>-entry_qnt.
        add ls_item-po_pr_qnt to <ls_item>-po_pr_qnt.
        add ls_item-amount_lc to <ls_item>-amount_lc.
        add ls_item-amount_sv to <ls_item>-amount_sv.
      else.
        insert ls_item into table lt_item.
      endif.
    endloop.

*--- gr item details
    loop at lt_item into ls_item.

*--- po details once only
      if rs_goodsreceipt_deep-po_number is initial.

*---- lookup further po details
        select single bsart waers
          from ekko
          into (l_bsart, rs_goodsreceipt_deep-currency)
          where ebeln = ls_item-po_number.
        if l_bsart <> ac_standard_po.
          rs_goodsreceipt_deep-mat_doc = '_'.
          exit.
        endif.

        rs_goodsreceipt_deep-po_number = ls_item-po_number.
        rs_goodsreceipt_deep-vendor = ls_item-vendor.

*---- vendor name
        select single name1
          from lfa1
          into rs_goodsreceipt_deep-vendor_name
          where lifnr = ls_item-vendor.

*--- lookup po line item details not supplied by BAPI above
        select ebelp loekz txz01 netpr mwskz webre effwr knttp
          from ekpo
          into corresponding fields of table lt_ekpo
          where ebeln = ls_item-po_number.

*--- is myfi flag (after legacy enablement all NB's are now 'MyFi')
        rs_goodsreceipt_deep-po_is_myfi = abap_true.

      endif.

*--- gr display item details
      clear ls_goodsreceiptitem.
      ls_goodsreceiptitem-selected = abap_true.
      ls_goodsreceiptitem-line_status = 'None'.
      ls_goodsreceiptitem-ref_doc_no = ls_header-ref_doc_no.
      ls_goodsreceiptitem-header_txt = ls_header-header_txt.
      ls_goodsreceiptitem-doc_date = ls_header-doc_date.
      ls_goodsreceiptitem-matdoc_itm = ls_item-matdoc_itm.
      ls_goodsreceiptitem-po_item = ls_item-po_item.
      ls_goodsreceiptitem-entry_uom = ls_item-entry_uom.
      ls_goodsreceiptitem-entry_qnt = ls_item-entry_qnt.

      read table lt_mseg into ls_mseg
        with table key
          zeile = ls_item-matdoc_itm.
      if sy-subrc = 0.
        if ls_mseg-shkzg ne 'S'.
          ls_goodsreceiptitem-entry_qnt = - ls_goodsreceiptitem-entry_qnt.
        endif.
      endif.

*--- check revesred
      select single mblnr
        from mseg
        into l_mblnr
        where smbln = i_mat_doc
          and sjahr = i_doc_year
          and smblp = ls_item-matdoc_itm.
      if sy-subrc = 0.
        ls_goodsreceiptitem-reversed = abap_true.
      endif.

*--- po line item details
      clear ls_ekpo.
      read table lt_ekpo into ls_ekpo
        with table key
          ebelp = ls_item-po_item.
      if sy-subrc = 0.
        ls_goodsreceiptitem-short_text = ls_ekpo-txz01.
        ls_goodsreceiptitem-net_price = ls_ekpo-netpr.
        ls_goodsreceiptitem-tax_code = ls_ekpo-mwskz.
      endif.

*--- exclusions: deleted/blocked lines, acct assignment asset unknown
      if ls_ekpo-knttp = 'X'
      or ls_ekpo-loekz is not initial.
        ls_goodsreceiptitem-no_more_gr = abap_true.
      else.

*--- check quantity remaining to ensure it can and flag if it can be canclled or not
*    NOTE: all lines are returned to the front end because the are always displayed in view
*    mode but hidden (by the app) in cancel mode
        clear lt_xekbes.
        clear lt_xekbez.
        call function 'ME_READ_HISTORY'
          exporting
            ebeln  = ls_item-po_number
            ebelp  = ls_item-po_item
            webre  = abap_true
          tables
            xekbes = lt_xekbes
            xekbez = lt_xekbez.
        if sy-subrc = 0.
*--- if gr based IV flag is set then check that:
*    (a) quantity remaining is equal to quantity to be cancelled, and
*    (b) no amount has been invoiced
          if ls_ekpo-webre = abap_true.
            read table lt_xekbez into ls_xekbez
              with key
                ebelp	= ls_item-po_item
                zekkn	= 00
                lfgja	= i_doc_year
                lfbnr = i_mat_doc
                lfpos	= ls_item-matdoc_itm.
            if sy-subrc = 0.
              if ls_xekbez-bpmng <> ls_item-entry_qnt
              or ls_xekbez-bprem is not initial.
                ls_goodsreceiptitem-no_more_gr = abap_true.
              endif.
            endif.
          else.
*--- if gr based IV flag is NOT set then check that the amount to be
*    cancelled is greater than gr - ir amount for at line level
*    (because we cannot check at material doc level with gr based IV)
            read table lt_xekbes into ls_xekbes
              with key
                ebelp	= ls_item-po_item
                zekkn	= 00.
            if sy-subrc = 0.
              if ls_goodsreceiptitem-entry_qnt > ( ls_xekbes-bpmng - ls_xekbes-bprem ).
                ls_goodsreceiptitem-no_more_gr = abap_true.
              endif.
            endif.
          endif.
        endif.
      endif.

      append ls_goodsreceiptitem to rs_goodsreceipt_deep-items.
    endloop.

*--- total po value
    loop at lt_ekpo into ls_ekpo where loekz ne 'L'.
      add ls_ekpo-effwr to rs_goodsreceipt_deep-total_value.
    endloop.

*--- total po value
    call function 'CURRENCY_AMOUNT_SAP_TO_BAPI'
      exporting
        currency    = rs_goodsreceipt_deep-currency
        sap_amount  = rs_goodsreceipt_deep-total_value
      importing
        bapi_amount = rs_goodsreceipt_deep-total_value.

  endif.

endmethod.


* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Private Method ZCL_ZLFC_GR_DPC_EXT->_READ_PO
* +-------------------------------------------------------------------------------------------------+

* | [<-()] RS_PURCHASEORDER_DEEP          TYPE        TY_PURCHASEORDER_DEEP

* | [!CX!] /IWBEP/CX_MGW_TECH_EXCEPTION

method _read_po.
*-----------------------------------------------------------------------------------------
* DESCRIPTION:  Helper Method to read Purchase Order details
*
* AUTHOR:       Fred Chelashaw
* CREATE DATE:  01.04.2020
*
* ACTIVITY:     AD-11-32-10 - MyFi Purchase Receipting
* DEV SCRIPT:   DS-11-32-10-01 - MyFi Purchase Receipting App
*-----------------------------------------------------------------------------------------
* VERSION CONTROL:
*
* Date       Author    CTS Req    SCR        Description
* ---------- --------- ---------- ---------- ---------------------------------------------
* 01.04.2020 220977FKC RD2K919103 6000001870 DFG BAR 2020-02 MyFi Purchase Receipting
* 21.10.2020 220977FKC RD2K919880 6000002064 VIM Integration with MyFi
* 30.11.2020 220977FKC RD2K920449 6000002172 MyFi Purchasing (Release 2.3)
*-----------------------------------------------------------------------------------------

  data:
    lo_message_container type ref to /iwbep/if_message_container,
    ls_purchaseorderitem type zcl_zlfc_gr_mpc=>ts_purchaseorderitem,
    ls_header            type bapimepoheader,
    l_ebeln              type ebeln,
    l_frgrl	             type frgrl,
    lt_return            type bapiret2_t,
    lt_item              type standard table of bapimepoitem,
    ls_item              like line of lt_item,
    lt_history           type standard table of bapiekbe,
    lt_history_total     type standard table of bapiekbes,
    ls_history_total     like line of lt_history_total,
    lt_po_line           type hashed table of ty_po_line
                           with unique key ebelp,
    ls_po_line           like line of lt_po_line,
    lt_tax_code          type hashed table of ty_tax_code
                           with unique key mwskz,
    ls_tax_code          like line of lt_tax_code,
    lt_vim_selected      type standard table of ty_vim_selected,
    ls_vim_selected      like line of lt_vim_selected.

*--- message container setup
  lo_message_container = mo_context->get_message_container( ).

*--- check if a vim document
  if strlen( i_po_number ) = 12.
    rs_purchaseorder_deep-vimdocid = i_po_number.
    select single ebeln xblnr gross_amount waers
      from /opt/vim_1head
      into (l_ebeln, rs_purchaseorder_deep-invnumber, rs_purchaseorder_deep-vim_invoice_amount, rs_purchaseorder_deep-vim_currency)
      where docid = rs_purchaseorder_deep-vimdocid.
  else.
    l_ebeln = i_po_number.
  endif.

*--- get po detail
  call function 'BAPI_PO_GETDETAIL1'
    exporting
      purchaseorder    = l_ebeln
    importing
      poheader         = ls_header
    tables
      return           = lt_return
      poitem           = lt_item
      pohistory        = lt_history
      pohistory_totals = lt_history_total.
  if sy-subrc <> 0.
    lo_message_container->add_messages_from_bapi(
      it_bapi_messages = lt_return
      iv_add_to_response_header = abap_true ).
  endif.
*--- if not 'NB' mark as excluded

  if ls_header-po_number is initial.

  elseif ls_header-doc_type <> ac_standard_po.
    rs_purchaseorder_deep-excluded = abap_true.
    rs_purchaseorder_deep-po_number = l_ebeln.
  else.

*--- header details from BAPI
    move-corresponding ls_header to rs_purchaseorder_deep.

*--- check if in held state
    if ls_header-memory ne abap_true.
      rs_purchaseorder_deep-complete = abap_true.
    endif.

*--- po release flag (not supplied by the above BAPI)
    select single frgrl
      from ekko
      into l_frgrl
      where ebeln = ls_header-po_number.

    if l_frgrl ne abap_true.
      rs_purchaseorder_deep-approved = abap_true.
    endif.

*--- get vendor name
    select single name1
      from lfa1
      into rs_purchaseorder_deep-vendor_name
      where lifnr = ls_header-vendor.

*---> is myfi flag (after legacy enablement all NB's are now 'MyFi')
    rs_purchaseorder_deep-is_my_fi = abap_true.

*--- lookup PO line item value (not supplied by the above BAPI)
    select ebelp effwr
      from ekpo
      into corresponding fields of table lt_po_line
      where ebeln = l_ebeln.

*--- lookup tax codes
    select tt~mwskz kk~kbetr
    from t007a as tt
      inner join a003 as aa
      on aa~mwskz = tt~mwskz
      inner join konp as kk
      on kk~knumh = aa~knumh and kk~kopos = 01
    into corresponding fields of table lt_tax_code
    where tt~kalsm = ac_taxau
      and tt~mwskz like 'P%'
      and ( aa~kschl = 'ZZ02' or aa~kschl = 'ZZ04' )
      and aa~aland = 'AU'.

*--- convert to entity format items
    loop at lt_item into ls_item.

*----> convert to entity item format
      clear ls_purchaseorderitem.

*---> line effective value
      read table lt_po_line into ls_po_line
        with table key
          ebelp = ls_item-po_item.
      if sy-subrc = 0.
        ls_purchaseorderitem-effective_value = ls_po_line-effwr.
        call function 'CURRENCY_AMOUNT_SAP_TO_BAPI'
          exporting
            currency    = rs_purchaseorder_deep-currency
            sap_amount  = ls_purchaseorderitem-effective_value
          importing
            bapi_amount = ls_purchaseorderitem-effective_value.
        if ls_item-delete_ind ne 'L'.
          add ls_po_line-effwr to rs_purchaseorder_deep-total_value.
        endif.
      endif.

*---> line exclusions:
      if ls_item-gr_ind is initial            " GR flag not set
      or ls_item-no_more_gr is not initial    " No more GR flag set
      or ls_item-delete_ind is not initial    " Deletion indicator
      or ls_item-final_inv is not initial     " Final invoice indicator
      or ls_item-acctasscat = 'X'.            " Asset Unknown Account Assignment
        continue.
      endif.

*----> convert to entity item format
      move-corresponding ls_item to ls_purchaseorderitem.

*----> is account assignment asset
      if ls_item-acctasscat = 'A'.

*-----> skip is user has no asset auth
        authority-check object 'A_B_ANLKL'
          id 'BUKRS' field rs_purchaseorder_deep-comp_code
          id 'ACTVT' field '01'.
        if sy-subrc <> 0.
          continue.
        else.

*-----> if user has auth set asset flags and load asset keys
          rs_purchaseorder_deep-has_asset = abap_true.
          ls_purchaseorderitem-is_asset = abap_true.

          select ebelp as po_item
              anln1 as asset_no
              anln2 as sub_number
            from ekkn
            appending corresponding fields of table rs_purchaseorder_deep-assets
            where ebeln = ls_header-po_number
              and ebelp = ls_item-po_item.

        endif.
      endif.

*----> calcualte quantity remaining
      loop at lt_history_total into ls_history_total where po_item = ls_item-po_item.
        add ls_history_total-deliv_qty to ls_purchaseorderitem-quantity_delivered.
      endloop.
      ls_purchaseorderitem-quantity_remaining = ls_purchaseorderitem-quantity - ls_purchaseorderitem-quantity_delivered.
      if ls_purchaseorderitem-quantity_remaining is initial.
        continue.
      endif.

*---> tax code
      read table lt_tax_code into ls_tax_code
        with table key
          mwskz = ls_item-tax_code.
      if sy-subrc = 0.
        ls_purchaseorderitem-tax_rate = ls_tax_code-kbetr / 1000.
      endif.

*----> append entity item
      append ls_purchaseorderitem to rs_purchaseorder_deep-items.

    endloop.

*--- total po value
    call function 'CURRENCY_AMOUNT_SAP_TO_BAPI'
      exporting
        currency    = rs_purchaseorder_deep-currency
        sap_amount  = rs_purchaseorder_deep-total_value
      importing
        bapi_amount = rs_purchaseorder_deep-total_value.

*---> if is VIM more details
    if rs_purchaseorder_deep-vimdocid is not initial.

*--- vim invoice amount
      call function 'CURRENCY_AMOUNT_SAP_TO_BAPI'
        exporting
          currency    = rs_purchaseorder_deep-vim_currency
          sap_amount  = rs_purchaseorder_deep-vim_invoice_amount
        importing
          bapi_amount = rs_purchaseorder_deep-vim_invoice_amount.

*      select taxamount valueextax
*        from zvim_gr_app_gr
*        into corresponding fields of table lt_vim_selected
*        where vimdocid = rs_purchaseorder_deep-vimdocid
*          and selectedgr = abap_true.
*
**--- calculate vim balance
*      rs_purchaseorder_deep-vim_balance = rs_purchaseorder_deep-vim_invoice_amount.
*      loop at lt_vim_selected into ls_vim_selected.
*        subtract ls_vim_selected-valueextax from rs_purchaseorder_deep-vim_balance.
*        subtract ls_vim_selected-taxamount from rs_purchaseorder_deep-vim_balance.
*      endloop.

    endif.

  endif.

endmethod.
